22  }

32  // projecting the relative velocity onto the normal vector.
33  // Since the normal vector is normalized, we just use the dot product
34  // (see: https://www.webmatematik.dk/lektioner/sarligt-for-htx/vektorer-i-planen/projektion-fra-enhedsvektor)
35  // Also eq. 2 from pdf
36  float contactSpeed = Vec2d::Dot(relativeVelocity, m.normal);

39  // In that case there is no need for collision resolution
40  if(contactSpeed < 0)

44  float e = m.entityA->restitution*m.entityB->restitution;
45 
46  // Create an impulse scalar (eq. 6 from pdf)
47  // NOTE: The formula has been changed from eq. 6 to use contact speed instead of relative velocity
48  // This makes it a litte simpler, but we have to use a square root for normalization, which inhibits performance
49  float j = -(1+e)*contactSpeed/(1/m.entityA->mass + 1/m.entityB->mass);
50 
51  // Eq. 8.a to calculate speed change from impulse
52  Vec2d impulseA = (m.normal * j)/m.entityA->mass;
53  Vec2d impulseB = (m.normal * j)/m.entityB->mass;
54 
55  // From newtons third law, an action has an opposite reaction, therefore
56  // when we add velocity to one object we remove from another
57  m.entityA->velocity -= impulseA;
58  m.entityB->velocity += impulseB;
59 
60  // Calculate friction
61  // Since we have updated the velocoties, we need to redo the earlier calculations
62  if(staticA) {
63  m.entityA->velocity = Vec2d();
64  }
65  if(staticB) {
66  m.entityB->velocity = Vec2d();
67  }
68  relativeVelocity = m.entityB->velocity - m.entityA->velocity;
69  contactSpeed = Vec2d::Dot(relativeVelocity, m.normal);
70 
71  // We create a tangent vector to the incident face, which descripes the friction force direction
72  Vec2d tangent = (relativeVelocity - m.normal * contactSpeed ).Normalized();
73 
74  // By projection the relative velocity onto the tangent we get the "sliding" speed or friction speed
75  float frictionSpeed = Vec2d::Dot(relativeVelocity, tangent);
76 
77  float staticFriction = sqrt(m.entityA->staticFriction * m.entityB->staticFriction);
78  float dynamicFriction = sqrt(m.entityA->dynamicFriction * m.entityB->dynamicFriction);
79 
80  // Calculate scalar for friction impulse
81  float fj = -frictionSpeed / (1/m.entityA->mass + 1/m.entityB->mass);
82 
83  if(abs(fj) < 0.001)
84  return;
85 
86  Vec2d friction;
87  if(abs(fj) < j * staticFriction)
88  friction = tangent * fj; // If the friction is small, then static friciton applies
89  else
90  friction = tangent * -j * dynamicFriction; // otherwise dynamic friciton
91 
92  // Apply friction impulses
93  if(!staticA) {
94  m.entityA->velocity += friction/m.entityA->mass;
95  }
96  if(!staticB) {
97  m.entityB->velocity -= friction/m.entityB->mass;
98  }
